function createHandler(value) {
	return function() {
		alert(`this is ${value}`);
	};
}

var elements = document.getElementsByTagName("button");
var n = elements.length;
for (var i = 0; i < n; i++) {
	elements[i].onclick = createHandler(i + 1);
}
